from flask import Flask, request, jsonify, render_template

import qrcode

import io

import base64

 

app = Flask(__name__)

 

inventory_items = {

"bl6630": {"description": "Blue LED Light", "location": "chattanooga", "quantity": 10},

"2010810hd": {"description": "High Definition Camera", "location": "nashville", "quantity": 5},

"slr950": {"description": "SLR Camera", "location": "truck 1", "quantity": 3},

"kmf505ar": {"description": "KMF505AR Part", "location": "chattanooga", "quantity": 8},

"kmf549ar": {"description": "KMF549AR Part", "location": "nashville", "quantity": 12},

}

 

user_roles = {

"admin": ["add_item", "change_part_number", "move_item"],

"user": ["view_inventory"]

}

 

def check_role(required_role):

def decorator(func):

def wrapper(*args, **kwargs):

user_role = request.headers.get('Role')

if user_role in user_roles and required_role in user_roles[user_role]:

return func(*args, **kwargs)

return jsonify({"message": "Access Denied"}), 403

return wrapper

return decorator

 

@app.route('/add_item', methods=['POST'])

@check_role('add_item')

def add_item():

data = request.json

part_number = data.get('part_number')

description = data.get('description')

location = data.get('location')

quantity = data.get('quantity')

 

if part_number in inventory_items:

return jsonify({"message": "Item already exists"}), 400

 

inventory_items[part_number] = {"description": description, "location": location, "quantity": quantity}

return jsonify({"message": "Item added successfully"}), 201

 

@app.route('/search_item', methods=['GET'])

def search_item():

part_number = request.args.get('part_number')

if part_number in inventory_items:

return jsonify(inventory_items[part_number]), 200

return jsonify({"message": "Item not found"}), 404

 

@app.route('/generate_qr', methods=['GET'])

def generate_qr():

part_number = request.args.get('part_number')

if part_number in inventory_items:

item_info = f"{part_number},{inventory_items[part_number]['description']},{inventory_items[part_number]['location']},{inventory_items[part_number]['quantity']}"

qr = qrcode.make(item_info)

 

buffered = io.BytesIO()

qr.save(buffered, format="PNG")

qr_base64 = base64.b64encode(buffered.getvalue()).decode('utf-8')

 

return jsonify({"qr_code": f"data:image/png;base64,{qr_base64}"}), 200

 

return jsonify({"message": "Item not found"}), 404

 

@app.route('/inventory', methods=['GET'])

@check_role('view_inventory')

def get_inventory():

location_filter = request.args.get('location')

if location_filter:

filtered_items = {k: v for k, v in inventory_items.items() if v["location"] == location_filter}

return jsonify(filtered_items), 200

 

return jsonify(inventory_items), 200

 

@app.route('/')

def home():

return render_template('index.html')

 

if __name__ == '__main__':

app.run(debug=True)
